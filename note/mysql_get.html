# MySQL-get資料

# 1. 取得SQL資料

資料夾：routes
檔案：member.js

# 路由get部分

router.get("/dashboard", function (req, res, next) {
  getinfotest(req, res);
  mySqlOrderListData(req, res);
  res.render("member_dashboard_0920honda.ejs", {
    sqlorderlist: sqlOrderListTestdata,
    sqlMenberdata : sqlMenberdata,
  });
});

概念

  1. 進入/dashboard路由之後，呼叫執行兩個function：getinfotest、mySqlOrderListData
  2. 執行res.render
  3. render內參數：
    1.要回傳的頁面member_dashboard_0920honda.ejs
    2.物件，包含屬性(ex:sqlorderlist)、值(ex:sqlOrderListTestdata）
  4. 物件內的值(sqlOrderListTestdata、sqlMenberdata，是由剛剛呼叫的兩個function產生

# 呼叫function來取得資料的部分

var sqlMenberdata =  [] ; 
function getinfotest(req, res) {
  req.mysql.query(
    "select * from member where memberId = 100938467",
    [],
    function (err, result1) {
      sqlMenberdata = result1 ;
      console.log(sqlMenberdata);
    }
  );
}

var sqlOrderListTestdata =  [] ; 
function mySqlOrderListData(req, res) {
  req.mysql.query(
    "SELECT * FROM normalorder  WHERE memberId = 100938467",
    [],
    function (err, result) {
      if (err) throw err;
      sqlOrderListTestdata =  result ; 
      console.log(sqlOrderListTestdata);
    }
    );
  }

概念

  1. 把要帶入render的值(ex:sqlMenberdata)宣告為空陣列
  2. function(ex:getinfotest)去抓MySQL資料，並存放在[ ]
  3. 透過function內的子function將結果(result1)存到剛剛宣到的陣列(sqlMenberdata)
